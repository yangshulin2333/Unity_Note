# P3-5：写 ResourceController

打开：

```text
StarterPlayer
└─ StarterPlayerScripts
   └─ Client
      └─ Controllers
         └─ ResourceController
```

如果没有就新建 `ModuleScript`。

替换成：

```lua
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")
local Workspace = game:GetService("Workspace")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)
local ResourceConfig = require(ReplicatedStorage.Shared.Config.ResourceConfig)

local ResourceController = {}

local initialized = false
local currentNode = nil
local isCollecting = false

local function getNodePart(node)
	if node:IsA("BasePart") then
		return node
	end

	if node:IsA("Model") then
		if node.PrimaryPart then
			return node.PrimaryPart
		end

		return node:FindFirstChildWhichIsA("BasePart", true)
	end

	return nil
end

local function getResourceType(node)
	local resourceType = node:GetAttribute("ResourceType")

	if resourceType then
		return resourceType
	end

	if node.Parent and node.Parent.Name == "Trees" then
		return "Tree"
	end

	if node.Parent and node.Parent.Name == "Rocks" then
		return "Rock"
	end

	return nil
end

local function getAllResourceNodes(resourceNodesFolder)
	local nodes = {}

	local treesFolder = resourceNodesFolder:FindFirstChild("Trees")
	local rocksFolder = resourceNodesFolder:FindFirstChild("Rocks")

	if treesFolder then
		for _, node in ipairs(treesFolder:GetChildren()) do
			table.insert(nodes, node)
		end
	end

	if rocksFolder then
		for _, node in ipairs(rocksFolder:GetChildren()) do
			table.insert(nodes, node)
		end
	end

	return nodes
end

function ResourceController.Init()
	if initialized then
		warn("[ResourceController] 已经初始化过，跳过重复初始化")
		return
	end
	initialized = true

	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

	local gatherPromptGui = playerGui:WaitForChild("GatherPromptGui", 10)
	if not gatherPromptGui then
		warn("[ResourceController] 找不到 GatherPromptGui")
		return
	end

	local promptRoot = gatherPromptGui:WaitForChild("Root", 10)
	local promptLabel = promptRoot:WaitForChild("PromptLabel", 10)

	local gatherProgressGui = playerGui:WaitForChild("GatherProgressGui", 10)
	if not gatherProgressGui then
		warn("[ResourceController] 找不到 GatherProgressGui")
		return
	end

	local progressRoot = gatherProgressGui:WaitForChild("Root", 10)
	local statusLabel = progressRoot:WaitForChild("StatusLabel", 10)
	local progressBar = progressRoot:WaitForChild("ProgressBar", 10)
	local fill = progressBar:WaitForChild("Fill", 10)

	local remotesFolder = ReplicatedStorage:WaitForChild("Remotes")
	local eventsFolder = remotesFolder:WaitForChild("Events")
	local resourceCollectEvent = eventsFolder:WaitForChild(RemoteNames.Events.ResourceCollectRequest)

	local interactables = Workspace:WaitForChild("Interactables")
	local resourceNodesFolder = interactables:WaitForChild("ResourceNodes")

	local function hidePrompt()
		promptRoot.Visible = false
		currentNode = nil
	end

	local function showPrompt(node)
		local resourceType = getResourceType(node)
		local config = resourceType and ResourceConfig.NodeTypes[resourceType]

		if not config then
			hidePrompt()
			return
		end

		currentNode = node
		promptLabel.Text = config.PromptText
		promptRoot.Visible = true
	end

	local function hideProgress()
		progressRoot.Visible = false
		fill.Size = UDim2.new(0, 0, 1, 0)
	end

	local function showProgress(text)
		progressRoot.Visible = true
		statusLabel.Text = text
		fill.Size = UDim2.new(0, 0, 1, 0)
	end

	local function runProgress(duration)
		local startTime = os.clock()

		while os.clock() - startTime < duration do
			local alpha = math.clamp((os.clock() - startTime) / duration, 0, 1)
			fill.Size = UDim2.new(alpha, 0, 1, 0)
			task.wait()
		end

		fill.Size = UDim2.new(1, 0, 1, 0)
	end

	local function startCollect()
		if isCollecting then
			return
		end

		if not currentNode then
			return
		end

		local node = currentNode
		local resourceType = getResourceType(node)
		local config = resourceType and ResourceConfig.NodeTypes[resourceType]

		if not config then
			warn("[ResourceController] 无效资源点:", node.Name)
			return
		end

		isCollecting = true
		promptRoot.Visible = false

		print("[ResourceController] 开始采集:", node.Name, resourceType)

		showProgress(config.CollectingText)
		runProgress(config.CollectTime)

		resourceCollectEvent:FireServer(resourceType, node.Name)
	end

	resourceCollectEvent.OnClientEvent:Connect(function(response)
		if not response then
			return
		end

		if response.success then
			statusLabel.Text = "+" .. tostring(response.amount) .. " " .. tostring(response.rewardType)
			print("[ResourceController] 采集成功:", response.rewardType, response.amount)
		else
			statusLabel.Text = response.message or "采集失败"
			warn("[ResourceController] 采集失败:", response.message)
		end

		task.delay(0.7, function()
			hideProgress()
			isCollecting = false
		end)
	end)

	RunService.RenderStepped:Connect(function()
		if isCollecting then
			return
		end

		local character = player.Character
		if not character then
			hidePrompt()
			return
		end

		local humanoidRootPart = character:FindFirstChild("HumanoidRootPart")
		if not humanoidRootPart then
			hidePrompt()
			return
		end

		local nearestNode = nil
		local nearestDistance = ResourceConfig.DetectDistance

		for _, node in ipairs(getAllResourceNodes(resourceNodesFolder)) do
			local nodePart = getNodePart(node)

			if nodePart then
				local distance = (humanoidRootPart.Position - nodePart.Position).Magnitude

				if distance <= nearestDistance then
					nearestDistance = distance
					nearestNode = node
				end
			end
		end

		if nearestNode then
			showPrompt(nearestNode)
		else
			hidePrompt()
		end
	end)

	UserInputService.InputBegan:Connect(function(input, gameProcessedEvent)
		if gameProcessedEvent then
			return
		end

		if input.KeyCode == Enum.KeyCode.E and currentNode and not isCollecting then
			startCollect()
		end
	end)

	print("[ResourceController] 初始化完成")
end

return ResourceController
```

# P3-7：写 ResourceSystem

打开：

```text
ServerScriptService
└─ Server
   └─ Systems
      └─ ResourceSystem
```

替换成：

```lua
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local ServerScriptService = game:GetService("ServerScriptService")
local Workspace = game:GetService("Workspace")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)
local ResourceConfig = require(ReplicatedStorage.Shared.Config.ResourceConfig)

local ServerFolder = ServerScriptService:WaitForChild("Server")
local ServicesFolder = ServerFolder:WaitForChild("Services")

local DataService = require(ServicesFolder:WaitForChild("DataService"))

local ResourceSystem = {}

local nodeCooldownEndTime = {}

local function getNodePart(node)
	if node:IsA("BasePart") then
		return node
	end

	if node:IsA("Model") then
		if node.PrimaryPart then
			return node.PrimaryPart
		end

		return node:FindFirstChildWhichIsA("BasePart", true)
	end

	return nil
end

local function findResourceNode(resourceType, nodeName)
	local interactables = Workspace:WaitForChild("Interactables")
	local resourceNodesFolder = interactables:WaitForChild("ResourceNodes")

	local folderName = nil

	if resourceType == "Tree" then
		folderName = "Trees"
	elseif resourceType == "Rock" then
		folderName = "Rocks"
	else
		return nil
	end

	local targetFolder = resourceNodesFolder:FindFirstChild(folderName)
	if not targetFolder then
		return nil
	end

	local node = targetFolder:FindFirstChild(nodeName)
	if not node then
		return nil
	end

	local attributeType = node:GetAttribute("ResourceType")
	if attributeType and attributeType ~= resourceType then
		return nil
	end

	return node
end

local function isPlayerCloseEnough(player, node)
	if not player.Character then
		return false
	end

	local humanoidRootPart = player.Character:FindFirstChild("HumanoidRootPart")
	if not humanoidRootPart then
		return false
	end

	local nodePart = getNodePart(node)
	if not nodePart then
		return false
	end

	local distance = (humanoidRootPart.Position - nodePart.Position).Magnitude

	return distance <= ResourceConfig.ServerCollectDistance
end

local function isNodeCoolingDown(node)
	local endTime = nodeCooldownEndTime[node]

	if not endTime then
		return false
	end

	return os.clock() < endTime
end

local function setNodeCooldown(node, seconds)
	nodeCooldownEndTime[node] = os.clock() + seconds
end

function ResourceSystem.Init()
	local remotesFolder = ReplicatedStorage:WaitForChild("Remotes")
	local eventsFolder = remotesFolder:WaitForChild("Events")

	local resourceCollectEvent = eventsFolder:WaitForChild(RemoteNames.Events.ResourceCollectRequest)

	resourceCollectEvent.OnServerEvent:Connect(function(player, resourceType, nodeName)
		print("[ResourceSystem] 收到采集请求:", player.Name, resourceType, nodeName)

		if type(resourceType) ~= "string" or type(nodeName) ~= "string" then
			warn("[ResourceSystem] 采集参数非法")
			resourceCollectEvent:FireClient(player, {
				success = false,
				message = "采集参数非法",
			})
			return
		end

		local config = ResourceConfig.NodeTypes[resourceType]
		if not config then
			warn("[ResourceSystem] 未知资源类型:", resourceType)
			resourceCollectEvent:FireClient(player, {
				success = false,
				message = "未知资源类型",
			})
			return
		end

		local node = findResourceNode(resourceType, nodeName)
		if not node then
			warn("[ResourceSystem] 找不到资源点:", resourceType, nodeName)
			resourceCollectEvent:FireClient(player, {
				success = false,
				message = "找不到资源点",
			})
			return
		end

		if isNodeCoolingDown(node) then
			warn("[ResourceSystem] 资源点冷却中:", nodeName)
			resourceCollectEvent:FireClient(player, {
				success = false,
				message = "资源点正在恢复中",
			})
			return
		end

		if not isPlayerCloseEnough(player, node) then
			warn("[ResourceSystem] 玩家距离资源点太远:", player.Name, nodeName)
			resourceCollectEvent:FireClient(player, {
				success = false,
				message = "距离太远",
			})
			return
		end

		local success, result = DataService.AddCurrency(player, config.RewardType, config.RewardAmount)

		if not success then
			warn("[ResourceSystem] 发放资源失败:", result)
			resourceCollectEvent:FireClient(player, {
				success = false,
				message = result,
			})
			return
		end

		setNodeCooldown(node, config.RespawnTime)

		print("[ResourceSystem] 采集成功:", player.Name, config.RewardType, config.RewardAmount)

		resourceCollectEvent:FireClient(player, {
			success = true,
			rewardType = config.RewardType,
			amount = config.RewardAmount,
			nodeName = nodeName,
		})
	end)

	print("[ResourceSystem] 初始化完成")
end

return ResourceSystem
```

# 这一步你最可能遇到的报错

## 1. `找不到 GatherProgressGui`

说明你没建：

```text
StarterGui > GatherProgressGui
```

或者名字拼错。

---

## 2. `找不到 ResourceCollectRequest`

说明：

- `RemoteNames` 里没有
    
- 或 `RemoteManager` 没创建
    
- 或 `RemoteManager.Init()` 没先执行
    

---

## 3. `找不到资源点`

检查：

```text
Workspace > Interactables > ResourceNodes > Trees > Tree_01
Workspace > Interactables > ResourceNodes > Rocks > Rock_01
```

以及资源点名字是否和客户端发的一样。

---

## 4. HUD 没刷新

检查 `DataService.AddCurrency()` 里有没有：

```lua
DataService.PushPlayerData(player)
```
