## Step 2：写代码（直接复制）

```lua
local Players = game:GetService("Players")

local HUDController = {}

function HUDController.Init()
	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

	local mainHUD = playerGui:WaitForChild("MainHUD")
	local root = mainHUD:WaitForChild("Root")
	local resourceBar = root:WaitForChild("ResourceBar")

	local goldLabel = resourceBar:WaitForChild("GoldLabel")
	local woodLabel = resourceBar:WaitForChild("WoodLabel")
	local stoneLabel = resourceBar:WaitForChild("StoneLabel")
	local levelLabel = resourceBar:WaitForChild("LevelLabel")

	print("[HUDController] 初始化完成")

	-- 👉 假数据
	local data = {
		Gold = 100,
		Wood = 20,
		Stone = 15,
		Level = 1
	}

	-- 👉 更新UI
	goldLabel.Text = "金币: " .. data.Gold
	woodLabel.Text = "木材: " .. data.Wood
	stoneLabel.Text = "石头: " .. data.Stone
	levelLabel.Text = "等级: " .. data.Level
end

return HUDController
```

---

## Step 2：写代码

```lua
local Players = game:GetService("Players")

local SelectionController = {}

function SelectionController.Init()
	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

	local gui = playerGui:WaitForChild("SelectLordGui")
	local root = gui:WaitForChild("Root")

	local choiceArea = root:WaitForChild("ChoiceArea")
	local weiButton = choiceArea:WaitForChild("WeiButton")
	local shuButton = choiceArea:WaitForChild("ShuButton")
	local wuButton = choiceArea:WaitForChild("WuButton")

	local selectedLabel = root:WaitForChild("SelectedLabel")
	local confirmButton = root:WaitForChild("ConfirmButton")

	local selectedFaction = nil

	-- 👉 打开UI（测试用）
	gui.Enabled = true

	print("[SelectionController] 初始化完成")

	weiButton.MouseButton1Click:Connect(function()
		selectedFaction = "Wei"
		selectedLabel.Text = "当前选择: 魏"
	end)

	shuButton.MouseButton1Click:Connect(function()
		selectedFaction = "Shu"
		selectedLabel.Text = "当前选择: 蜀"
	end)

	wuButton.MouseButton1Click:Connect(function()
		selectedFaction = "Wu"
		selectedLabel.Text = "当前选择: 吴"
	end)

	confirmButton.MouseButton1Click:Connect(function()
		if not selectedFaction then
			print("❌ 请先选择阵营")
			return
		end

		print("✅ 已选择阵营:", selectedFaction)
	end)
end

return SelectionController
```
