## `UIManager` 代码

把 `UIManager` 内容替换成：

```lua
local Players = game:GetService("Players")

local UIManager = {}

local initialized = false
local registeredWindows = {}

local function getRoot(gui)
	if not gui then
		return nil
	end

	return gui:FindFirstChild("Root")
end

function UIManager.Init()
	if initialized then
		warn("[UIManager] 已经初始化过，跳过重复初始化")
		return
	end
	initialized = true

	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

	local windowGuiNames = {
		"SignInGui",
		"RechargeGui",
		"FriendGui",
		"ShopGui",
		"InventoryGui",
		"QuestGui",
		"MailGui",
		"SettingsGui",
	}

	for _, guiName in ipairs(windowGuiNames) do
		local gui = playerGui:FindFirstChild(guiName)

		if gui then
			local root = getRoot(gui)

			if root then
				registeredWindows[guiName] = {
					Gui = gui,
					Root = root,
				}

				root.Visible = false

				print("[UIManager] 注册窗口:", guiName)
			else
				warn("[UIManager] 找到 Gui 但没有 Root:", guiName)
			end
		else
			warn("[UIManager] 未找到 Gui，跳过注册:", guiName)
		end
	end

	print("[UIManager] 初始化完成")
end

function UIManager.Open(guiName)
	local windowInfo = registeredWindows[guiName]

	if not windowInfo then
		warn("[UIManager] 未注册窗口，无法打开:", guiName)
		return
	end

	UIManager.CloseAll()

	windowInfo.Root.Visible = true

	print("[UIManager] 打开窗口:", guiName)
end

function UIManager.Close(guiName)
	local windowInfo = registeredWindows[guiName]

	if not windowInfo then
		warn("[UIManager] 未注册窗口，无法关闭:", guiName)
		return
	end

	windowInfo.Root.Visible = false

	print("[UIManager] 关闭窗口:", guiName)
end

function UIManager.Toggle(guiName)
	local windowInfo = registeredWindows[guiName]

	if not windowInfo then
		warn("[UIManager] 未注册窗口，无法切换:", guiName)
		return
	end

	if windowInfo.Root.Visible then
		UIManager.Close(guiName)
	else
		UIManager.Open(guiName)
	end
end

function UIManager.CloseAll()
	for guiName, windowInfo in pairs(registeredWindows) do
		windowInfo.Root.Visible = false
	end
end

function UIManager.BindCloseButton(guiName)
	local windowInfo = registeredWindows[guiName]

	if not windowInfo then
		warn("[UIManager] 未注册窗口，无法绑定关闭按钮:", guiName)
		return
	end

	local root = windowInfo.Root
	local window = root:FindFirstChild("Window")

	if not window then
		warn("[UIManager] 窗口缺少 Window:", guiName)
		return
	end

	local header = window:FindFirstChild("Header")

	if not header then
		warn("[UIManager] 窗口缺少 Header:", guiName)
		return
	end

	local closeButton = header:FindFirstChild("CloseButton")

	if not closeButton then
		warn("[UIManager] 窗口缺少 CloseButton:", guiName)
		return
	end

	closeButton.MouseButton1Click:Connect(function()
		UIManager.Close(guiName)
	end)

	print("[UIManager] 绑定关闭按钮:", guiName)
end

function UIManager.BindAllCloseButtons()
	for guiName, _ in pairs(registeredWindows) do
		UIManager.BindCloseButton(guiName)
	end
end

return UIManager
```

---

# 这段代码你只需要先理解 4 个点

## 1. `registeredWindows`

这是一个表，用来记录有哪些窗口：

```lua
registeredWindows["SignInGui"] = {
	Gui = SignInGui,
	Root = SignInGui.Root,
}
```

以后打开窗口就不用每次重新找。

---

## 2. `UIManager.Open("SignInGui")`

它会：

```text
先关闭所有窗口
再打开 SignInGui.Root
```

---

## 3. `UIManager.Close("SignInGui")`

它会：

```text
把 SignInGui.Root.Visible = false
```

---

## 4. `BindCloseButton`

它会自动找到：

```text
Root > Window > Header > CloseButton
```

然后给它绑定点击关闭。

所以你的弹窗结构一定要统一。

## `MainMenuController` 代码

把内容替换成：

```lua
local Players = game:GetService("Players")

local MainMenuController = {}

local initialized = false

local function findButton(root, pathList)
	local current = root

	for _, childName in ipairs(pathList) do
		current = current:FindFirstChild(childName)

		if not current then
			return nil
		end
	end

	return current
end

local function bindButton(button, callback, debugName)
	if not button then
		warn("[MainMenuController] 找不到按钮，跳过绑定:", debugName)
		return
	end

	button.MouseButton1Click:Connect(callback)

	print("[MainMenuController] 已绑定按钮:", debugName)
end

function MainMenuController.Init(UIManager)
	if initialized then
		warn("[MainMenuController] 已经初始化过，跳过重复初始化")
		return
	end
	initialized = true

	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

	local mainHUD = playerGui:WaitForChild("MainHUD")
	local root = mainHUD:WaitForChild("Root")

	print("[MainMenuController] 初始化完成")

	local signInButton = findButton(root, {
		"TopBar",
		"TopButtons",
		"SignInButton",
	})

	local rechargeButton = findButton(root, {
		"TopBar",
		"TopButtons",
		"RechargeButton",
	})

	local mailButton = findButton(root, {
		"TopBar",
		"TopButtons",
		"MailButton",
	})

	local friendButton = findButton(root, {
		"RightMenu",
		"FriendButton",
	})

	local questButton = findButton(root, {
		"RightMenu",
		"QuestButton",
	})

	local settingsButton = findButton(root, {
		"RightMenu",
		"SettingsButton",
	})

	local shopButton = findButton(root, {
		"BottomMenu",
		"ShopButton",
	})

	local inventoryButton = findButton(root, {
		"BottomMenu",
		"InventoryButton",
	})

	local dungeonButton = findButton(root, {
		"BottomMenu",
		"DungeonButton",
	})

	bindButton(signInButton, function()
		UIManager.Open("SignInGui")
	end, "SignInButton")

	bindButton(rechargeButton, function()
		UIManager.Open("RechargeGui")
	end, "RechargeButton")

	bindButton(friendButton, function()
		UIManager.Open("FriendGui")
	end, "FriendButton")

	bindButton(shopButton, function()
		UIManager.Open("ShopGui")
	end, "ShopButton")

	bindButton(inventoryButton, function()
		UIManager.Open("InventoryGui")
	end, "InventoryButton")

	bindButton(mailButton, function()
		UIManager.Open("MailGui")
	end, "MailButton")

	bindButton(questButton, function()
		UIManager.Open("QuestGui")
	end, "QuestButton")

	bindButton(settingsButton, function()
		UIManager.Open("SettingsGui")
	end, "SettingsButton")

	bindButton(dungeonButton, function()
		print("[MainMenuController] 点击 DungeonButton，副本界面后续接入")
	end, "DungeonButton")
end

return MainMenuController
```

# 这段代码你先理解 3 件事

## 1. `findButton`

它是安全查找按钮。

比如：

```lua
findButton(root, {
	"TopBar",
	"TopButtons",
	"SignInButton",
})
```

等于去找：

```text
Root > TopBar > TopButtons > SignInButton
```

如果中间某一级不存在，它不会直接爆红，而是返回 nil。

---

## 2. `bindButton`

它负责绑定按钮点击。

如果按钮不存在，它会 warn，不会让整个客户端崩掉。

这对你现在很有用，因为你可能 UI 还没全部建完。

---

## 3. `MainMenuController.Init(UIManager)`

这里把 `UIManager` 传进来，是因为：

```text
MainMenuController 负责监听按钮
UIManager 负责打开关闭窗口
```

这就是分工。

这样主界面按钮才有管理器可用。

# 第 10 步：常见问题排查

## 问题 1：点按钮没反应

看 Output 有没有：

```text
[MainMenuController] 找不到按钮，跳过绑定
```

如果有，说明路径不对。

例如代码找的是：

```text
Root > TopBar > TopButtons > SignInButton
```

但你实际可能放成了：

```text
Root > TopButtons > SignInButton
```

这时你要么改 UI 层级，要么改代码路径。

---

## 问题 2：按钮绑定成功，但窗口打不开

看 Output 有没有：

```text
[UIManager] 未注册窗口，无法打开: SignInGui
```

如果有，说明：

- `SignInGui` 不在 `StarterGui`
    
- 或运行后没有进入 `PlayerGui`
    
- 或 `SignInGui` 里没有 `Root`
    

---

## 问题 3：窗口打开了，但看不见

检查：

```text
SignInGui.Enabled = true
Root.Visible = true
Window.Visible = true
Window.Size 不为 0
Window.Position 在屏幕内
```

---

## 问题 4：CloseButton 点了没反应

看 Output 有没有：

```text
[UIManager] 窗口缺少 Window
[UIManager] 窗口缺少 Header
[UIManager] 窗口缺少 CloseButton
```

说明弹窗结构没有按统一模板建。
