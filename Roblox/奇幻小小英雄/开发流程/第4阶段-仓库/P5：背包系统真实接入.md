你的要求我先固化成规则：

1. **UI 资源你自己找**：我只告诉你需要哪些 UI 对象、类型、名字、放在哪里。
    
2. **每次创建对象必须说明数据结构**：比如 `Frame / TextButton / ImageLabel / TextLabel`。
    
3. **不跳步骤**：先 UI 层级，再配置，再 Remote，再服务端，再客户端，再测试。
    
4. **第一版只做装备背包**：材料背包、出售装备、强化装备后面再做。
    
5. **金币增长暂时不处理**：你说后续打算通过“卖东西”获得金币，这个合理，但属于后面的 `SellItemSystem`，P5 先不做。
    

P5 的意义很明确：你已经能通过商店购买 `木剑 / 铁剑 / 铁甲 / 布甲`，现在要让这些装备真正进入背包并显示出来。开发文档里也明确提到基础装备系统需要做到“玩家能持有装备、能穿戴装备、装备影响属性”，并且 MVP 阶段背包 UI 的目标是“查看已有装备、穿戴/卸下”。

function DataService.FindInventoryItem(player, instanceId)
	local data = DataService.GetPlayerData(player)

	if not data then
		return nil, nil
	end

	if type(data.Inventory) ~= "table" then
		return nil, data
	end

	for _, item in ipairs(data.Inventory) do

## 第 4 步：增加穿戴装备函数

```lua
function DataService.EquipItem(player, instanceId)
	local item, data = DataService.FindInventoryItem(player, instanceId)

	if not data then
		return false, "找不到玩家数据"
	end

	if not item then
		return false, "背包中没有该装备"
	end

	local equipment = EquipmentConfig.Equipments[item.ItemId]
	if not equipment then
		return false, "装备配置不存在: " .. tostring(item.ItemId)
	end

	local slot = equipment.Slot
	if not slot then
		return false, "装备没有槽位配置"
	end

	if type(data.Equipped) ~= "table" then
		data.Equipped = {
			Weapon = nil,
			Armor = nil,
		}
	end

	data.Equipped[slot] = instanceId

	DataService.PushPlayerData(player)

	return true, {
		Slot = slot,
		Item = item,
	}
end
```

---

## 第 5 步：增加卸下装备函数

```lua
function DataService.UnequipSlot(player, slot)
	local data = DataService.GetPlayerData(player)

	if not data then
		return false, "找不到玩家数据"
	end

	if type(data.Equipped) ~= "table" then
		data.Equipped = {
			Weapon = nil,
			Armor = nil,
		}
	end

	if data.Equipped[slot] == nil then
		return false, "该槽位没有装备"
	end

	data.Equipped[slot] = nil

	DataService.PushPlayerData(player)

	return true, {
		Slot = slot,
	}
end
```

# P5-7：新增 InventorySystem 服务端系统

位置：

```text
ServerScriptService
└─ Server
   └─ Systems
      └─ InventorySystem
```

新建 `ModuleScript`：

```text
InventorySystem
```

代码：

```lua
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local ServerScriptService = game:GetService("ServerScriptService")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)

local ServerFolder = ServerScriptService:WaitForChild("Server")
local ServicesFolder = ServerFolder:WaitForChild("Services")

local DataService = require(ServicesFolder:WaitForChild("DataService"))

local InventorySystem = {}

local function sendEquipResult(equipEvent, player, response)
	equipEvent:FireClient(player, response)
end

local function sendUnequipResult(unequipEvent, player, response)
	unequipEvent:FireClient(player, response)
end

function InventorySystem.Init()
	local remotesFolder = ReplicatedStorage:WaitForChild("Remotes")
	local eventsFolder = remotesFolder:WaitForChild("Events")

	local equipItemEvent = eventsFolder:WaitForChild(RemoteNames.Events.EquipItemRequest)
	local unequipItemEvent = eventsFolder:WaitForChild(RemoteNames.Events.UnequipItemRequest)

	equipItemEvent.OnServerEvent:Connect(function(player, instanceId)
		print("[InventorySystem] 收到穿戴请求:", player.Name, instanceId)

		if type(instanceId) ~= "string" then
			sendEquipResult(equipItemEvent, player, {
				success = false,
				message = "装备参数错误",
			})
			return
		end

		local success, result = DataService.EquipItem(player, instanceId)

		if not success then
			warn("[InventorySystem] 穿戴失败:", result)
			sendEquipResult(equipItemEvent, player, {
				success = false,
				message = result,
			})
			return
		end

		print("[InventorySystem] 穿戴成功:", player.Name, result.Slot)

		sendEquipResult(equipItemEvent, player, {
			success = true,
			message = "穿戴成功",
			slot = result.Slot,
			instanceId = instanceId,
			itemId = result.Item.ItemId,
		})
	end)

	unequipItemEvent.OnServerEvent:Connect(function(player, slot)
		print("[InventorySystem] 收到卸下请求:", player.Name, slot)

		if type(slot) ~= "string" then
			sendUnequipResult(unequipItemEvent, player, {
				success = false,
				message = "槽位参数错误",
			})
			return
		end

		local success, result = DataService.UnequipSlot(player, slot)

		if not success then
			warn("[InventorySystem] 卸下失败:", result)
			sendUnequipResult(unequipItemEvent, player, {
				success = false,
				message = result,
			})
			return
		end

		print("[InventorySystem] 卸下成功:", player.Name, result.Slot)

		sendUnequipResult(unequipItemEvent, player, {
			success = true,
			message = "卸下成功",
			slot = result.Slot,
		})
	end)

	print("[InventorySystem] 初始化完成")
end

return InventorySystem
```

位置：

   └─ Client

新建 `ModuleScript`：

InventoryController

代码如下：

```lua
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)
local EquipmentConfig = require(ReplicatedStorage.Shared.Config.EquipmentConfig)

local InventoryController = {}

local initialized = false

local latestPlayerData = nil
local selectedItem = nil
local selectedCard = nil

local ui = {}

local function waitChild(parent, childName)
	local child = parent:WaitForChild(childName, 5)

	if not child then
		warn("[InventoryController] 找不到:", childName)
	end

	return child
end

local function keepLayoutObject(child, template)
	return child == template
		or child:IsA("UIGridLayout")
		or child:IsA("UIListLayout")
		or child:IsA("UIPadding")
		or child:IsA("UICorner")
		or child:IsA("UIStroke")
end

local function clearOldItemCards()
	for _, child in ipairs(ui.ItemScroll:GetChildren()) do
		if not keepLayoutObject(child, ui.Template) then
			child:Destroy()
		end
	end
end

local function getEquipment(itemId)
	return EquipmentConfig.Equipments[itemId]
end

local function getSlotDisplayName(slot)
	if EquipmentConfig.SlotDisplayNames and EquipmentConfig.SlotDisplayNames[slot] then
		return EquipmentConfig.SlotDisplayNames[slot]
	end

	return slot or "未知槽位"
end

local function findInventoryItemByInstanceId(instanceId)
	if not latestPlayerData or type(latestPlayerData.Inventory) ~= "table" then
		return nil
	end

	for _, item in ipairs(latestPlayerData.Inventory) do
		if item.InstanceId == instanceId then
			return item
		end
	end

	return nil
end

local function getEquippedSlotByInstanceId(instanceId)
	if not latestPlayerData or type(latestPlayerData.Equipped) ~= "table" then
		return nil
	end

	for slot, equippedInstanceId in pairs(latestPlayerData.Equipped) do
		if equippedInstanceId == instanceId then
			return slot
		end
	end

	return nil
end

local function isItemEquipped(instanceId)
	return getEquippedSlotByInstanceId(instanceId) ~= nil
end

local function setSelectedVisual(card, selected)
	local selectedFrame = card:FindFirstChild("SelectedFrame")
	if selectedFrame then
		selectedFrame.Visible = selected
	end
end

local function setEquippedTag(card, visible)
	local equippedTag = card:FindFirstChild("EquippedTag")

	if equippedTag then
		equippedTag.Visible = visible

		if equippedTag:IsA("TextLabel") then
			equippedTag.Text = "已装备"
		end
	end
end

local function setTip(text)
	if ui.TipLabel then
		ui.TipLabel.Text = text
	end
end

local function buildStatsText(equipment)
	if not equipment then
		return "无属性"
	end

	local lines = {}

	if equipment.Attack and equipment.Attack ~= 0 then
		table.insert(lines, "攻击 +" .. tostring(equipment.Attack))
	end

	if equipment.Defense and equipment.Defense ~= 0 then
		table.insert(lines, "防御 +" .. tostring(equipment.Defense))
	end

	if equipment.Hp and equipment.Hp ~= 0 then
		table.insert(lines, "生命 +" .. tostring(equipment.Hp))
	end

	if #lines == 0 then
		return "无属性"
	end

	return table.concat(lines, "\n")
end

local function updateDetailPanel(item)
	selectedItem = item

	if not item then
		if ui.ItemNameLabel then
			ui.ItemNameLabel.Text = "未选择装备"
		end

		if ui.ItemDescLabel then
			ui.ItemDescLabel.Text = "请选择左侧背包中的装备。"
		end

		if ui.StatsLabel then
			ui.StatsLabel.Text = ""
		end

		if ui.DetailLevelLabel then
			ui.DetailLevelLabel.Text = ""
		end

		if ui.EquipButton then
			ui.EquipButton.Visible = false
		end

		if ui.UnequipButton then
			ui.UnequipButton.Visible = false
		end

		return
	end

	local equipment = getEquipment(item.ItemId)
	local equippedSlot = getEquippedSlotByInstanceId(item.InstanceId)

	if ui.ItemNameLabel then
		ui.ItemNameLabel.Text = equipment and equipment.DisplayName or item.ItemId
	end

	if ui.ItemDescLabel then
		ui.ItemDescLabel.Text = equipment and equipment.Desc or "装备配置不存在"
	end

	if ui.StatsLabel then
		ui.StatsLabel.Text = buildStatsText(equipment)
	end

	if ui.DetailLevelLabel then
		ui.DetailLevelLabel.Text = "强化等级 +" .. tostring(item.Level or 0)
	end

	if ui.EquipButton then
		ui.EquipButton.Visible = equippedSlot == nil
	end

	if ui.UnequipButton then
		ui.UnequipButton.Visible = equippedSlot ~= nil
	end
end

local function updateEquippedPanel()
	if not latestPlayerData then
		return
	end

	local equipped = latestPlayerData.Equipped or {}

	local function updateSlot(slotName, slotFrame)
		if not slotFrame then
			return
		end

		local itemNameLabel = slotFrame:FindFirstChild("ItemNameLabel")
		local slotNameLabel = slotFrame:FindFirstChild("SlotNameLabel")

		if slotNameLabel then
			slotNameLabel.Text = getSlotDisplayName(slotName)
		end

		local instanceId = equipped[slotName]
		if not instanceId then
			if itemNameLabel then
				itemNameLabel.Text = "未装备"
			end
			return
		end

		local item = findInventoryItemByInstanceId(instanceId)
		if not item then
			if itemNameLabel then
				itemNameLabel.Text = "数据异常"
			end
			return
		end

		local equipment = getEquipment(item.ItemId)
		if itemNameLabel then
			itemNameLabel.Text = equipment and equipment.DisplayName or item.ItemId
		end
	end

	updateSlot("Weapon", ui.WeaponSlot)
	updateSlot("Armor", ui.ArmorSlot)
end

local function fillItemCard(card, item)
	local equipment = getEquipment(item.ItemId)

	card.Name = item.InstanceId
	card.Visible = true
	card.Text = ""

	local icon = card:FindFirstChild("Icon")
	if icon and equipment and equipment.Icon and equipment.Icon ~= "" then
		icon.Image = equipment.Icon
	end

	local nameLabel = card:FindFirstChild("NameLabel")
	if nameLabel then
		nameLabel.Text = equipment and equipment.DisplayName or item.ItemId
	end

	local levelLabel = card:FindFirstChild("LevelLabel")
	if levelLabel then
		levelLabel.Text = "+" .. tostring(item.Level or 0)
	end

	local slotLabel = card:FindFirstChild("SlotLabel")
	if slotLabel then
		slotLabel.Text = equipment and getSlotDisplayName(equipment.Slot) or "未知"
	end

	setSelectedVisual(card, false)
	setEquippedTag(card, isItemEquipped(item.InstanceId))
end

local function renderInventory()
	if not latestPlayerData then
		return
	end

	clearOldItemCards()
	updateEquippedPanel()

	local inventory = latestPlayerData.Inventory or {}

	if #inventory == 0 then
		setTip("背包暂无装备")
		updateDetailPanel(nil)
		return
	end

	for _, item in ipairs(inventory) do
		local card = ui.Template:Clone()
		card.Parent = ui.ItemScroll

		fillItemCard(card, item)

		card.MouseButton1Click:Connect(function()
			if selectedCard then
				setSelectedVisual(selectedCard, false)
			end

			selectedCard = card
			setSelectedVisual(card, true)
			updateDetailPanel(item)
			setTip("已选择装备")
		end)
	end

	if selectedItem then
		local stillExists = findInventoryItemByInstanceId(selectedItem.InstanceId)
		if stillExists then
			updateDetailPanel(stillExists)
		else
			selectedItem = nil
			selectedCard = nil
			updateDetailPanel(nil)
		end
	else
		updateDetailPanel(nil)
	end
end

function InventoryController.Init()
	if initialized then
		warn("[InventoryController] 已经初始化过，跳过重复初始化")
		return
	end
	initialized = true

	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

	local inventoryGui = waitChild(playerGui, "InventoryGui")
	if not inventoryGui then return end

	local root = waitChild(inventoryGui, "Root")
	if not root then return end

	local window = waitChild(root, "Window")
	if not window then return end

	local content = waitChild(window, "Content")
	if not content then return end

	ui.EquippedPanel = waitChild(content, "EquippedPanel")
	ui.ItemScroll = waitChild(content, "ItemScroll")
	ui.DetailPanel = waitChild(content, "DetailPanel")

	if not ui.EquippedPanel or not ui.ItemScroll or not ui.DetailPanel then
		warn("[InventoryController] InventoryGui 结构不完整")
		return
	end

	ui.WeaponSlot = ui.EquippedPanel:FindFirstChild("WeaponSlot")
	ui.ArmorSlot = ui.EquippedPanel:FindFirstChild("ArmorSlot")

	ui.Template = waitChild(ui.ItemScroll, "InventoryItemTemplate")
	if not ui.Template then return end
	ui.Template.Visible = false

	ui.ItemNameLabel = ui.DetailPanel:FindFirstChild("ItemNameLabel")
	ui.ItemDescLabel = ui.DetailPanel:FindFirstChild("ItemDescLabel")
	ui.StatsLabel = ui.DetailPanel:FindFirstChild("StatsLabel")
	ui.DetailLevelLabel = ui.DetailPanel:FindFirstChild("LevelLabel")
	ui.EquipButton = ui.DetailPanel:FindFirstChild("EquipButton")
	ui.UnequipButton = ui.DetailPanel:FindFirstChild("UnequipButton")

	local footer = window:FindFirstChild("Footer")
	ui.TipLabel = footer and footer:FindFirstChild("TipLabel")

	local remotesFolder = ReplicatedStorage:WaitForChild("Remotes")
	local eventsFolder = remotesFolder:WaitForChild("Events")

	local playerDataUpdateEvent = eventsFolder:WaitForChild(RemoteNames.Events.PlayerDataUpdate)
	local equipItemEvent = eventsFolder:WaitForChild(RemoteNames.Events.EquipItemRequest)
	local unequipItemEvent = eventsFolder:WaitForChild(RemoteNames.Events.UnequipItemRequest)

	if ui.EquipButton then
		ui.EquipButton.MouseButton1Click:Connect(function()
			if not selectedItem then
				setTip("请先选择装备")
				return
			end

			setTip("正在穿戴...")
			print("[InventoryController] 请求穿戴:", selectedItem.InstanceId)
			equipItemEvent:FireServer(selectedItem.InstanceId)
		end)
	end

	if ui.UnequipButton then
		ui.UnequipButton.MouseButton1Click:Connect(function()
			if not selectedItem then
				setTip("请先选择装备")
				return
			end

			local slot = getEquippedSlotByInstanceId(selectedItem.InstanceId)
			if not slot then
				setTip("该装备未穿戴")
				return
			end

			setTip("正在卸下...")
			print("[InventoryController] 请求卸下:", slot)
			unequipItemEvent:FireServer(slot)
		end)
	end

	playerDataUpdateEvent.OnClientEvent:Connect(function(data)
		latestPlayerData = data
		renderInventory()
	end)

	equipItemEvent.OnClientEvent:Connect(function(response)
		if not response then return end

		if response.success then
			setTip("穿戴成功")
			print("[InventoryController] 穿戴成功:", response.slot)
		else
			setTip("穿戴失败：" .. tostring(response.message))
			warn("[InventoryController] 穿戴失败:", response.message)
		end
	end)

	unequipItemEvent.OnClientEvent:Connect(function(response)
		if not response then return end

		if response.success then
			setTip("卸下成功")
			print("[InventoryController] 卸下成功:", response.slot)
		else
			setTip("卸下失败：" .. tostring(response.message))
			warn("[InventoryController] 卸下失败:", response.message)
		end
	end)

	updateDetailPanel(nil)

	print("[InventoryController] 初始化完成")
end

return InventoryController
```
