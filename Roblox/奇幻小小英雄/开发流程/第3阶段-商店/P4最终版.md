# 十、DataService 增加 3 个函数

你现在的 `DataService` 继续保留。

只需要补这几个能力。

## 1. 扣货币

```lua
function DataService.SpendCurrency(player, currencyName, amount)
	local data = DataService.GetPlayerData(player)

	if not data then
		return false, "找不到玩家数据"
	end

	if data[currencyName] == nil then
		return false, "不存在的货币字段: " .. tostring(currencyName)
	end

	if data[currencyName] < amount then
		return false, "资源不足"
	end

	data[currencyName] -= amount

	DataService.PushPlayerData(player)

	return true, data
end
```

## 2. 生成背包物品 ID

放在 `DataService` 顶部附近：

```lua
local nextInventoryId = 0

local function generateInventoryInstanceId(player, itemId)
	nextInventoryId += 1
	return tostring(player.UserId) .. "_" .. itemId .. "_" .. tostring(os.time()) .. "_" .. tostring(nextInventoryId)
end
```

---

## 3. 加入背包

```lua
function DataService.AddInventoryItem(player, itemId)
	local data = DataService.GetPlayerData(player)

	if not data then
		return false, "找不到玩家数据"
	end

	if type(data.Inventory) ~= "table" then
		data.Inventory = {}
	end

	local itemInstance = {
		InstanceId = generateInventoryInstanceId(player, itemId),
		ItemId = itemId,
		Level = 0,
	}

	table.insert(data.Inventory, itemInstance)

	DataService.PushPlayerData(player)

	return true, itemInstance
end
```

# 十一、ShopSystem：服务端购买裁判

新建或修改：

```text
ServerScriptService > Server > Systems > ShopSystem
```

```lua
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local ServerScriptService = game:GetService("ServerScriptService")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)
local ShopConfig = require(ReplicatedStorage.Shared.Config.ShopConfig)
local EquipmentConfig = require(ReplicatedStorage.Shared.Config.EquipmentConfig)

local ServerFolder = ServerScriptService:WaitForChild("Server")
local ServicesFolder = ServerFolder:WaitForChild("Services")

local DataService = require(ServicesFolder:WaitForChild("DataService"))

local ShopSystem = {}

local function sendResult(buyItemEvent, player, response)
	buyItemEvent:FireClient(player, response)
end

function ShopSystem.Init()
	local remotesFolder = ReplicatedStorage:WaitForChild("Remotes")
	local eventsFolder = remotesFolder:WaitForChild("Events")

	local buyItemEvent = eventsFolder:WaitForChild(RemoteNames.Events.BuyItemRequest)

	buyItemEvent.OnServerEvent:Connect(function(player, itemId)
		print("[ShopSystem] 收到购买请求:", player.Name, itemId)

		if type(itemId) ~= "string" then
			sendResult(buyItemEvent, player, {
				success = false,
				message = "商品参数错误",
			})
			return
		end

		local shopItem = ShopConfig.Items[itemId]
		if not shopItem then
			sendResult(buyItemEvent, player, {
				success = false,
				message = "商品不存在",
			})
			return
		end

		local equipment = EquipmentConfig.Equipments[shopItem.ItemId]
		if not equipment then
			sendResult(buyItemEvent, player, {
				success = false,
				message = "装备配置不存在",
			})
			return
		end

		local spendSuccess, spendMessage = DataService.SpendCurrency(
			player,
			shopItem.PriceType,
			shopItem.Price
		)

		if not spendSuccess then
			sendResult(buyItemEvent, player, {
				success = false,
				message = spendMessage,
			})
			return
		end

		local addSuccess, itemInstance = DataService.AddInventoryItem(player, shopItem.ItemId)

		if not addSuccess then
			sendResult(buyItemEvent, player, {
				success = false,
				message = "加入背包失败",
			})
			return
		end

		print("[ShopSystem] 购买成功:", player.Name, shopItem.ItemId)

		sendResult(buyItemEvent, player, {
			success = true,
			message = "购买成功",
			itemId = shopItem.ItemId,
			displayName = equipment.DisplayName,
			instanceId = itemInstance.InstanceId,
		})
	end)

	print("[ShopSystem] 初始化完成")
end

return ShopSystem
```

# 十二、ShopController：保留你的模板思路，重写成更清晰版

这个版本适配你现在的 UI：

```text
ShopItemTemplate
├─ SelectedFrame
├─ PriceButton
├─ CardBG
├─ Icon
├─ NameLabel
└─ DescLabel
```

```lua
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)
local ShopConfig = require(ReplicatedStorage.Shared.Config.ShopConfig)

local ShopController = {}

local initialized = false
local selectedButton = nil

local function waitChild(parent, childName)
	local child = parent:WaitForChild(childName, 5)

	if not child then
		warn("[ShopController] 找不到:", childName)
	end

	return child
end

local function keepLayoutObject(child, template)
	return child == template
		or child:IsA("UIListLayout")
		or child:IsA("UIGridLayout")
		or child:IsA("UIPadding")
		or child:IsA("UICorner")
		or child:IsA("UIStroke")
end

local function clearOldItems(itemScroll, template)
	for _, child in ipairs(itemScroll:GetChildren()) do
		if not keepLayoutObject(child, template) then
			child:Destroy()
		end
	end
end

local function setSelected(button, isSelected)
	local selectedFrame = button:FindFirstChild("SelectedFrame")
	if selectedFrame then
		selectedFrame.Visible = isSelected
	end
end

local function fillItemCard(card, itemConfig)
	card.Name = itemConfig.ItemId .. "Item"
	card.Visible = true
	card.Text = ""

	local nameLabel = card:FindFirstChild("NameLabel")
	if nameLabel then
		nameLabel.Text = itemConfig.DisplayName
	end

	local descLabel = card:FindFirstChild("DescLabel")
	if descLabel then
		descLabel.Text = itemConfig.Desc or ""
	end

	local priceButton = card:FindFirstChild("PriceButton")
	if priceButton then
		priceButton.Text = "购买 " .. tostring(itemConfig.Price) .. " " .. tostring(itemConfig.PriceType)
	end

	local icon = card:FindFirstChild("Icon")
	if icon and itemConfig.Icon and itemConfig.Icon ~= "" then
		icon.Image = itemConfig.Icon
	end

	setSelected(card, false)
end

function ShopController.Init()
	if initialized then
		warn("[ShopController] 已经初始化过，跳过重复初始化")
		return
	end
	initialized = true

	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

	local shopGui = waitChild(playerGui, "ShopGui")
	if not shopGui then return end

	local root = waitChild(shopGui, "Root")
	if not root then return end

	local window = waitChild(root, "Window")
	if not window then return end

	local content = waitChild(window, "Content")
	if not content then return end

	local itemScroll = waitChild(content, "ItemScroll")
	if not itemScroll then return end

	local template = waitChild(itemScroll, "ShopItemTemplate")
	if not template then return end

	local footer = window:FindFirstChild("Footer")
	local tipLabel = footer and footer:FindFirstChild("TipLabel")

	local remotesFolder = ReplicatedStorage:WaitForChild("Remotes")
	local eventsFolder = remotesFolder:WaitForChild("Events")
	local buyItemEvent = eventsFolder:WaitForChild(RemoteNames.Events.BuyItemRequest)

	template.Visible = false
	clearOldItems(itemScroll, template)

	local function setTip(text)
		if tipLabel then
			tipLabel.Text = text
		end
	end

	local function bindItemCard(card, itemConfig)
		card.MouseButton1Click:Connect(function()
			if selectedButton then
				setSelected(selectedButton, false)
			end

			selectedButton = card
			setSelected(card, true)

			setTip("已选择：" .. itemConfig.DisplayName)
		end)

		local priceButton = card:FindFirstChild("PriceButton")
		if priceButton then
			priceButton.MouseButton1Click:Connect(function()
				setTip("正在购买：" .. itemConfig.DisplayName)

				print("[ShopController] 请求购买:", itemConfig.ItemId)
				buyItemEvent:FireServer(itemConfig.ItemId)
			end)
		end
	end

	for _, itemId in ipairs(ShopConfig.DisplayOrder) do
		local itemConfig = ShopConfig.Items[itemId]

		if itemConfig then
			local card = template:Clone()
			card.Parent = itemScroll

			fillItemCard(card, itemConfig)
			bindItemCard(card, itemConfig)
		end
	end

	buyItemEvent.OnClientEvent:Connect(function(response)
		if not response then
			return
		end

		if response.success then
			setTip("购买成功：" .. tostring(response.displayName))
			print("[ShopController] 购买成功:", response.displayName)
		else
			setTip("购买失败：" .. tostring(response.message))
			warn("[ShopController] 购买失败:", response.message)
		end
	end)

	print("[ShopController] 初始化完成")
end

return ShopController
```

# 十四、ShopInteractController：按 E 打开商店

你说你的交互方式是对 Part 按 E，那么建议单独做这个控制器。

你先在场景里放商店交互点：

```text
Workspace
└─ Interactables
   └─ ShopPoints
      └─ MainShop
```

`MainShop` 是一个 Part：

```text
Anchored = true
CanCollide = false
Transparency = 0.5
```

然后新建：

```text
StarterPlayerScripts > Client > Controllers > ShopInteractController
```

简单版逻辑：

```lua
local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")
local Workspace = game:GetService("Workspace")

local ShopInteractController = {}

local initialized = false
local currentShopPoint = nil
local DETECT_DISTANCE = 12

function ShopInteractController.Init(UIManager)
	if initialized then
		return
	end
	initialized = true

	local player = Players.LocalPlayer

	local interactables = Workspace:WaitForChild("Interactables")
	local shopPoints = interactables:WaitForChild("ShopPoints")

	RunService.RenderStepped:Connect(function()
		local character = player.Character
		if not character then
			currentShopPoint = nil
			return
		end

		local humanoidRootPart = character:FindFirstChild("HumanoidRootPart")
		if not humanoidRootPart then
			currentShopPoint = nil
			return
		end

		currentShopPoint = nil

		for _, point in ipairs(shopPoints:GetChildren()) do
			if point:IsA("BasePart") then
				local distance = (humanoidRootPart.Position - point.Position).Magnitude

				if distance <= DETECT_DISTANCE then
					currentShopPoint = point
					break
				end
			end
		end
	end)

	UserInputService.InputBegan:Connect(function(input, gameProcessedEvent)
		if gameProcessedEvent then
			return
		end

		if input.KeyCode == Enum.KeyCode.E and currentShopPoint then
			print("[ShopInteractController] 打开商店:", currentShopPoint.Name)
			UIManager.Open("ShopGui")
		end
	end)

	print("[ShopInteractController] 初始化完成")
end

return ShopInteractController
```

这个暂时没做提示 UI，因为你已经有很多提示系统了。后面可以统一成一个 `InteractionPromptController`。
