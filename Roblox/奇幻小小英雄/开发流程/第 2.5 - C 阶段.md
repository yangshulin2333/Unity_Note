## 第 1 步：先做一个最小版 DataService

不是持久化，只是**内存版玩家数据**。

把 `ServerScriptService > Server > Services > DataService` 改成这样：

```lua
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local GameConstants = require(ReplicatedStorage.Shared.Constants.GameConstants)

local DataService = {}

local playerDataMap = {}

local function deepCopy(original)
	local copy = {}

	for key, value in pairs(original) do
		if type(value) == "table" then
			local nestedCopy = {}
			for nestedKey, nestedValue in pairs(value) do
				nestedCopy[nestedKey] = nestedValue
			end
			copy[key] = nestedCopy
		else
			copy[key] = value
		end
	end

	return copy
end

function DataService.Init()
	Players.PlayerAdded:Connect(function(player)
		playerDataMap[player.UserId] = deepCopy(GameConstants.DefaultPlayerData)
		print("[DataService] 已创建玩家数据:", player.Name)
	end)

	Players.PlayerRemoving:Connect(function(player)
		playerDataMap[player.UserId] = nil
		print("[DataService] 已移除玩家数据:", player.Name)
	end)

	print("[DataService] 初始化完成")
end

function DataService.GetPlayerData(player)
	return playerDataMap[player.UserId]
end

function DataService.SetFaction(player, faction)
	local data = playerDataMap[player.UserId]
	if not data then
		return false, "找不到玩家数据"
	end

	data.Faction = faction
	data.HasSelectedLord = true
	data.LordId = faction .. "_Lord"

	return true, data
end

return DataService
```

## 第 2 步：做服务端 `SelectionSystem`

把 `ServerScriptService > Server > Systems > SelectionSystem` 改成这样：

```lua
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local ServerScriptService = game:GetService("ServerScriptService")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)
local GameConstants = require(ReplicatedStorage.Shared.Constants.GameConstants)

local ServerFolder = ServerScriptService:WaitForChild("Server")
local ServicesFolder = ServerFolder:WaitForChild("Services")

local DataService = require(ServicesFolder:WaitForChild("DataService"))

local SelectionSystem = {}

function SelectionSystem.Init()
	local remotesFolder = ReplicatedStorage:WaitForChild("Remotes")
	local eventsFolder = remotesFolder:WaitForChild("Events")

	local selectLordEvent = eventsFolder:WaitForChild(RemoteNames.Events.SelectLordRequest)

	selectLordEvent.OnServerEvent:Connect(function(player, faction)
		print("[SelectionSystem] 收到选主公请求:", player.Name, faction)

		if type(faction) ~= "string" then
			warn("[SelectionSystem] faction 不是字符串")
			return
		end

		local validFaction =
			faction == GameConstants.Factions.Wei or
			faction == GameConstants.Factions.Shu or
			faction == GameConstants.Factions.Wu

		if not validFaction then
			warn("[SelectionSystem] 非法阵营:", faction)
			return
		end

		local data = DataService.GetPlayerData(player)
		if not data then
			warn("[SelectionSystem] 玩家数据不存在:", player.Name)
			return
		end

		if data.HasSelectedLord then
			warn("[SelectionSystem] 玩家已经选过主公:", player.Name)
			return
		end

		local success, result = DataService.SetFaction(player, faction)
		if not success then
			warn("[SelectionSystem] 设置阵营失败:", result)
			return
		end

		print("[SelectionSystem] 选主公成功:", player.Name, faction)
		selectLordEvent:FireClient(player, {
			success = true,
			faction = faction,
		})
	end)

	print("[SelectionSystem] 初始化完成")
end

return SelectionSystem
```

local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local RemoteNames = require(ReplicatedStorage.Shared.Constants.RemoteNames)

local SelectionController = {}

local initialized = false

function SelectionController.Init()
	if initialized then
		warn("[SelectionController] 已经初始化过，跳过重复初始化")
		return
	end
	initialized = true

	local player = Players.LocalPlayer
	local playerGui = player:WaitForChild("PlayerGui")

		selectedFaction = "Wei"

	end)

		selectedFaction = "Shu"

		selectedFaction = "Wu"
		selectedLabel.Text = "当前选择: 吴"
	end)

		if not selectedFaction then
			print("❌ 请先选择阵营")
			return
		end

		print("[SelectionController] 发送选主公请求:", selectedFaction)
		selectLordEvent:FireServer(selectedFaction)
	end)
end

return SelectionController
