## 第 16 步：编写 DataService

打开 `ServerScriptService > Services > DataService`

替换为下面代码：

```lua
local Players = game:GetService("Players")
local ServerStorage = game:GetService("ServerStorage")

local DefaultProfile = require(ServerStorage:WaitForChild("Data"):WaitForChild("DefaultProfile"))

local DataService = {}
DataService.Profiles = {}

local function deepCopy(original)
	local copy = {}
	for key, value in pairs(original) do
		if type(value) == "table" then
			copy[key] = deepCopy(value)
		else
			copy[key] = value
		end
	end
	return copy
end

function DataService:LoadProfile(player)
	local profile = deepCopy(DefaultProfile)
	profile.Player.Name = player.Name
	profile.Timers.LastLogin = os.time()

	self.Profiles[player] = profile
	return profile
end

function DataService:GetProfile(player)
	return self.Profiles[player]
end

function DataService:SaveProfile(player)
	local profile = self.Profiles[player]
	if profile then
		profile.Timers.LastSave = os.time()
		print("[DataService] Saved profile for", player.Name)
	end
end

function DataService:ReleaseProfile(player)
	self:SaveProfile(player)
	self.Profiles[player] = nil
end

return DataService
```

## 第 17 步：编写 BaseService

打开 `ServerScriptService > Services > BaseService`

替换为下面代码：

```lua
local Workspace = game:GetService("Workspace")

local BaseService = {}
BaseService.AssignedPlots = {}

local plotsFolder = Workspace:WaitForChild("Plots")

function BaseService:GetAllPlots()
	return plotsFolder:GetChildren()
end

function BaseService:GetPlotByPlayer(player)
	return self.AssignedPlots[player]
end

function BaseService:IsPlotFree(plot)
	for _, assignedPlot in pairs(self.AssignedPlots) do
		if assignedPlot == plot then
			return false
		end
	end
	return true
end

function BaseService:AssignPlot(player)
	if self.AssignedPlots[player] then
		return self.AssignedPlots[player]
	end

	for _, plot in ipairs(self:GetAllPlots()) do
		if self:IsPlotFree(plot) then
			self.AssignedPlots[player] = plot
			print("[BaseService] Assigned", plot.Name, "to", player.Name)
			return plot
		end
	end

	warn("[BaseService] No free plot for", player.Name)
	return nil
end

function BaseService:ReleasePlot(player)
	local plot = self.AssignedPlots[player]
	if plot then
		print("[BaseService] Released", plot.Name, "from", player.Name)
		self.AssignedPlots[player] = nil
	end
end

function BaseService:GetPlotSpawn(plot)
	if not plot then
		return nil
	end
	return plot:FindFirstChild("BaseSpawn")
end

return BaseService
```

## 第 18 步：编写 PlayerService
打开 `ServerScriptService > Services > PlayerService`

替换为下面代码：

```lua
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local DataService = require(script.Parent:WaitForChild("DataService"))
local BaseService = require(script.Parent:WaitForChild("BaseService"))

local remotesFolder = ReplicatedStorage:WaitForChild("Shared"):WaitForChild("Remotes")
local initClientRemote = remotesFolder:WaitForChild("InitClient")
local notifyPlotAssignedRemote = remotesFolder:WaitForChild("NotifyPlotAssigned")
local requestPlayerDataFunction = remotesFolder:WaitForChild("RequestPlayerData")

local PlayerService = {}

function PlayerService:OnPlayerAdded(player)
	local profile = DataService:LoadProfile(player)
	local plot = BaseService:AssignPlot(player)

	player.CharacterAdded:Connect(function(character)
		self:OnCharacterAdded(player, character)
	end)

	initClientRemote:FireClient(player, {
		PlayerName = profile.Player.Name,
		Level = profile.Player.Level,
		Currency = profile.Currency,
	})

	if plot then
		notifyPlotAssignedRemote:FireClient(player, plot.Name)
	end
end

function PlayerService:OnCharacterAdded(player, character)
	local plot = BaseService:GetPlotByPlayer(player)
	if not plot then
		return
	end

	local spawnPart = BaseService:GetPlotSpawn(plot)
	if not spawnPart then
		return
	end

	local hrp = character:WaitForChild("HumanoidRootPart", 5)
	if hrp then
		hrp.CFrame = spawnPart.CFrame + Vector3.new(0, 3, 0)
	end
end

function PlayerService:OnPlayerRemoving(player)
	BaseService:ReleasePlot(player)
	DataService:ReleaseProfile(player)
end

function PlayerService:Init()
	Players.PlayerAdded:Connect(function(player)
		self:OnPlayerAdded(player)
	end)

	Players.PlayerRemoving:Connect(function(player)
		self:OnPlayerRemoving(player)
	end)

	requestPlayerDataFunction.OnServerInvoke = function(player)
		local profile = DataService:GetProfile(player)
		if not profile then
			return nil
		end

		local plot = BaseService:GetPlotByPlayer(player)
		return {
			Player = profile.Player,
			Currency = profile.Currency,
			PlotName = plot and plot.Name or "None",
		}
	end
end

return PlayerService
```

```lua
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local controllersFolder = script.Parent:WaitForChild("Controllers")
local UIController = require(controllersFolder:WaitForChild("UIController"))
local PlotController = require(controllersFolder:WaitForChild("PlotController"))

local remotesFolder = ReplicatedStorage:WaitForChild("Shared"):WaitForChild("Remotes")

local notifyPlotAssignedRemote = remotesFolder:WaitForChild("NotifyPlotAssigned")
local requestPlayerDataFunction = remotesFolder:WaitForChild("RequestPlayerData")

UIController:Init()
PlotController:Init()

initClientRemote.OnClientEvent:Connect(function(data)
	UIController:UpdateInitData(data)
end)

notifyPlotAssignedRemote.OnClientEvent:Connect(function(plotName)
	PlotController:SetPlot(plotName)
end)

local success, result = pcall(function()
	return requestPlayerDataFunction:InvokeServer()
end)

if success and result then
	print("[ClientMain] Initial player data:", result)
else
	warn("[ClientMain] Failed to get initial player data")
end
```
